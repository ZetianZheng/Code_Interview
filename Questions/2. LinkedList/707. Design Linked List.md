class ListNode {
    int val;
    ListNode next;
    ListNode() {}
    
    ListNode(int val, ListNode next) {
        this.val = val;
        this.next = next;
    }

    ListNode(int val) {this.val = val;}

}
class MyLinkedList {
    // 单链表
    // 记录长度和虚拟头结点
    int size;
    ListNode head;

    public MyLinkedList() {
        this.size = 0;
        this.head = new ListNode();
    }
    
    // get the element of index in linkedlist
    public int get(int index) {
        ListNode curr = head;

        if (index < 0 || index >= size) {
            return -1;
        }

        // head 是虚拟头结点，所以index+1
        for (int i = 0; i <= index; i++) {
            curr = curr.next;
        }

        return curr.val;
    }
    
    public void addAtHead(int val) {
        addAtIndex(0, val);
    }
    
    public void addAtTail(int val) {
        addAtIndex(size, val);
    }
    
    // add the element val to index in the linkedlist
    public void addAtIndex(int index, int val) {
        // Add a node of value val before the indexth node in the linked list
        // need a prev pointer and curr pointer
        ListNode pre = head;

        if (index < 0) {
            index = 0;
        }
        
        if (index > size) {
            return;
        }
        

        this.size++;

        // get the node before indexth
        for (int i = 0; i < index; i++) {
            pre = pre.next;
        }

        // add new node:
        ListNode newNode = new ListNode(val, pre.next);
        pre.next = newNode;
        
    }
    
    
    public void deleteAtIndex(int index) {
        // delete at index, consider about index = 0, add a dummy node, 
        // use a pre node to record previous ndoe, a curr node record current one
        ListNode pre = head;

        if (index < 0 || index >= size) {
           return;
        }

        // get the node at indexth
        for (int i = 0; i < index; i++) {
            pre = pre.next;
        }

        pre.next = pre.next.next;
        this.size--;
    }
}

/**
 * Your MyLinkedList object will be instantiated and called as such:
 * MyLinkedList obj = new MyLinkedList();
 * int param_1 = obj.get(index);
 * obj.addAtHead(val);
 * obj.addAtTail(val);
 * obj.addAtIndex(index,val);
 * obj.deleteAtIndex(index);
